name: mentionheightinmeters;
parameters: person;

name: mentionheightinfeet;
parameters: person;
preconditions: !visibleproperty(0 height*);
creates: 0<height*;
effects: visibleproperty(0 height*) in(0<height* feet*) ;

name: mentionweight;
parameters: person;
preconditions: !visibleproperty(0 weight*);
creates: 0<weight*;
effects: visibleproperty(0 weight*) in(0<weight* kilograms*);

name: convertfeettometers;
parameters: distancevalue*;
preconditions: in (0 feet*);
creates: ;
effects: in(0 meters*) !in(0 feet*);
~
%VARIABLES
%NODES
@o,/0/,(number-0.3048),/0/,*
%RESULTS
meters,/0/
~

name: convertmeterstofeet;
parameters: distancevalue*;
preconditions: in(0 meters*);
creates: ;
effects: in(0 feet*) !in(0 meters*);
~
%VARIABLES
%NODES
@o,/0/,(number-0.3048),/0/,/
%RESULTS

~

name: computebmi;
parameters: person;
preconditions: visibleproperty(0 height*) visibleproperty(0 weight*) !visibleproperty(0 bmi*) in (0<height* meters*) in(0<weight* kilograms*);

effects: visibleproperty(0 bmi*) completed();
~
%VARIABLES
squared-number
%NODES
o,/0<height*/,/0<height*/,[squared],*,1
@o,[squared],/0<weight*/,/0<bmi*/,/
%RESULTS

~
